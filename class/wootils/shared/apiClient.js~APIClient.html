    public flattenEndpoints(endpoints: Object, parent: String): Object source

    Taks a dictionary of endpoints and flatten them on a single level.

    Params:

    NameTypeAttributeDescription
    endpoints Object

    A dictionary of named endpoints.
    parent String
    • optional
    • default: ''

    The parent key of the received endpoints. This is used when the method is calling itself recursively.

    Return:

    Object

    Example:

    console.log(APIClient.flattenEndpoints({
  endpointOne: 'endpoint-one',
  endpointTwo: {
    path: 'endpoint-two',
    query: {
      count: 20,
    },
  },
  endpointThree: {
    subEndpointThreeOne: 'sub-endpoint-three-one',
    subEndpointThreeTwo: {
      path: 'sub-endpoint-three-two',
      query: {
        count: 20,
      },
    },
  },
}));
// Will output
{
  endpointOne: 'endpoint-one',
  endpointTwo: {
    path: 'endpoint-two',
    query: {
      count: 20,
    },
  },
  'endpointThree.subEndpointThreeOne': 'sub-endpoint-three-one',
  'endpointThree.subEndpointThreeTwo': {
      path: 'sub-endpoint-three-two',
      query: {
        count: 20,
      },
    },
  },
}
