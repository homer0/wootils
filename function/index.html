    public appConfiguration(appName: string, defaultConfiguration: Object, options: Object): Provider source 

    import {appConfiguration} from 'wootils/wootils/node/appConfiguration.js'

    Generates a Provider with an already defined name, default configuration and options.

    Params:

    NameTypeAttributeDescription
    appName string
    • optional

    The name of the app.
    defaultConfiguration Object
    • optional

    The service default configuration.
    options Object
    • optional

    Options to customize the service.

    Return:

    Provider

    Example:

    // Generate the provider
const provider = appConfiguration('my-app', {
  birthday: '25-09-2015',
});
// Register it on the container
container.register(provider);
// Getting access to the service instance
const appConfiguration = container.get('appConfiguration');
